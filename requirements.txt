from fpdf import FPDF

class CustomPDF(FPDF):
    def header(self):
        # Only add header on the second page and onwards
        if self.page_no() > 1:
            self.set_font('Arial', 'B', 12)
            self.cell(0, 10, 'Scan Report', 0, 1, 'C')

    def footer(self):
        self.set_y(-15)
        self.set_font('Arial', 'I', 8)
        self.cell(0, 10, f'Page {self.page_no()}', 0, 0, 'C')

def create_pdf(results, filename, background_image_path):
    pdf = CustomPDF()
    
    # Add cover page with background image
    pdf.add_page()
    pdf.image(background_image_path, x=0, y=0, w=210, h=297)  # A4 size

    # Start text content from the second page
    pdf.add_page()
    
    pdf.set_left_margin(20)
    pdf.set_right_margin(20)
    pdf.set_top_margin(20)
    
    heading_color = (79, 129, 189)

    def check_and_add_page(required_space):
        """Check if there's enough space for the required content and add a new page if not."""
        remaining_space = pdf.h - pdf.get_y() - pdf.b_margin
        if remaining_space < required_space:
            pdf.add_page()

    for scan_name, content in results.items():
        # Check if there is enough space for the heading
        pdf.set_font('Arial', 'B', 14)
        pdf.set_text_color(*heading_color)
        heading_height = pdf.font_size * 2 + 10  # Estimate heading height
        check_and_add_page(heading_height)
        
        pdf.cell(0, 10, scan_name, ln=True, align='L')
        
        # Check if there is enough space for the content
        pdf.set_font('Arial', '', 12)
        pdf.set_text_color(0, 0, 0)
        lines = content.split('\n')
        line_height = pdf.font_size * 1.5
        content_height = len(lines) * line_height + 10  # Estimate content height
        check_and_add_page(content_height)

        # Replace dashes with dots
        content = '\n'.join([f"â€¢ {line[2:]}" if line.strip().startswith("- ") else line for line in lines])
        pdf.multi_cell(0, 10, content)
        
        # Add space after content
        pdf.ln(10)

    # Output the generated PDF
    pdf.output(filename)
    print(f"PDF report created: {filename}")
